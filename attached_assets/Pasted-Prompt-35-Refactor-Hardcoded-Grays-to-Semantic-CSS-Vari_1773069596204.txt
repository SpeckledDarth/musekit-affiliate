Problem

Fix — Replace Hardcoded Grays with Semantic Classes

import { createBrowserClient, createAdminClient } from "@musekit/database";
let _browserClient: ReturnType<typeof createBrowserClient> | null = null;
function getBrowserClient() {
  if (_browserClient) return _browserClient;
  try {
    _browserClient = createBrowserClient();
    return _browserClient;
  } catch (error) {
    console.error("Affiliate: Failed to create Supabase client:", error);
    throw error;
  }
}
export const supabase = getBrowserClient();
export function getServiceClient() {
  try {
    return createAdminClient();
  } catch (error) {
    console.error("Affiliate: Failed to create Supabase admin client:", error);
    throw error;
  }
}
